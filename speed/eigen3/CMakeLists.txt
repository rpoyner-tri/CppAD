# Build the speed/eigen3 directory tests
# Inherit build type from ../CMakeList.txt

# assert eigen3_prefix is defined
assert ( eigen3_prefix )

# Adds flags to the compiler command line for sources in the current directory
# and below. This command can be used to add any flags, but it was originally
# intended to add preprocessor definitions.
ADD_DEFINITIONS("-DCPPAD_EIGEN3_SPEED")

# Local include directories to search (not in package_prefix/includdir)
INCLUDE_DIRECTORIES( ${CMAKE_CURRENT_SOURCE_DIR}/../src )

# add_executable(<name> [WIN32] [MACOSX_BUNDLE] [EXCLUDE_FROM_ALL]
#                 source1 source2 ... sourceN
# )
SET(source_list ../main.cpp
    det_lu.cpp
    det_minor.cpp
    mat_mul.cpp
    ode.cpp
    poly.cpp
    sparse_hessian.cpp
    sparse_jacobian.cpp
)
set_compile_flags( speed_eigen3 "${cppad_debug_which}" "${source_list}" )
#
ADD_EXECUTABLE( speed_eigen3 EXCLUDE_FROM_ALL ${source_list} )

# eigen3 is an include file only library
TARGET_LINK_LIBRARIES(speed_eigen3 speed_src )

# Add the check_speed_eigen3 target
ADD_CUSTOM_TARGET(check_speed_eigen3
    speed_eigen3 correct 54321
    DEPENDS speed_eigen3 speed_src
)
MESSAGE(STATUS "make check_speed_eigen3: available")

# Change check depends in parent environment
add_to_list(check_speed_depends check_speed_eigen3)
SET(check_speed_depends "${check_speed_depends}" PARENT_SCOPE)
